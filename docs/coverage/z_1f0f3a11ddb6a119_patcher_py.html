22class SysModulesSaver(object): 

23 """Class that captures some subset of the current state of 

24 sys.modules. Pass in an iterator of module names to the 

25 constructor.""" 

26 

27 def __init__(self, module_names=()): 

28 self._saved = {} 

29 imp.acquire_lock() 

30 self.save(*module_names) 

31 

32 def save(self, *module_names): 

33 """Saves the named modules to the object.""" 

34 for modname in module_names: 

35 self._saved[modname] = sys.modules.get(modname, None) 

36 

37 def restore(self): 

38 """Restores the modules that the saver knows about into 

39 sys.modules. 

40 """ 

41 try: 

42 for modname, mod in six.iteritems(self._saved): 

43 if mod is not None: 

44 sys.modules[modname] = mod 

45 else: 

46 try: 

47 del sys.modules[modname] 

48 except KeyError: 

49 pass 

50 finally: 

51 imp.release_lock()

54def inject(module_name, new_globals, *additional_modules): 

55 """Base method for "injecting" greened modules into an imported module. It 

56 imports the module specified in *module_name*, arranging things so 

57 that the already-imported modules in *additional_modules* are used when 

58 *module_name* makes its imports. 

59 

60 **Note:** This function does not create or change any sys.modules item, so 

61 if your greened module use code like 'sys.modules["your_module_name"]', you 

62 need to update sys.modules by yourself. 

63 

64 *new_globals* is either None or a globals dictionary that gets populated 

65 with the contents of the *module_name* module. This is useful when creating 

66 a "green" version of some other module. 

67 

68 *additional_modules* should be a collection of two-element tuples, of the 

69 form (<name>, <module>). If it's not specified, a default selection of 

70 name/module pairs is used, which should cover all use cases but may be 

71 slower because there are inevitably redundant or unnecessary imports. 

72 """ 

73 patched_name = '__patched_module_' + module_name 

74 if patched_name in sys.modules: 

75 # returning already-patched module so as not to destroy existing 

76 # references to patched modules 

77 return sys.modules[patched_name] 

78 

79 if not additional_modules: 

80 # supply some defaults 

81 additional_modules = ( 

82 _green_os_modules() + 

83 _green_select_modules() + 

84 _green_socket_modules() + 

85 _green_thread_modules() + 

86 _green_time_modules()) 

87 # _green_MySQLdb()) # enable this after a short baking-in period 

88 

89 # after this we are gonna screw with sys.modules, so capture the 

90 # state of all the modules we're going to mess with, and lock 

91 saver = SysModulesSaver([name for name, m in additional_modules]) 

92 saver.save(module_name) 

93 

94 # Cover the target modules so that when you import the module it 

95 # sees only the patched versions 

96 for name, mod in additional_modules: 

97 sys.modules[name] = mod 

98 

99 # Remove the old module from sys.modules and reimport it while 

100 # the specified modules are in place 

101 sys.modules.pop(module_name, None) 

102 # Also remove sub modules and reimport. Use copy the keys to list 

103 # because of the pop operations will change the content of sys.modules 

104 # within th loop 

105 for imported_module_name in list(sys.modules.keys()): 

106 if imported_module_name.startswith(module_name + '.'): 

107 sys.modules.pop(imported_module_name, None) 

108 try: 

109 module = __import__(module_name, {}, {}, module_name.split('.')[:-1]) 

110 

111 if new_globals is not None: 

112 # Update the given globals dictionary with everything from this new module 

113 for name in dir(module): 

114 if name not in __exclude: 

115 new_globals[name] = getattr(module, name) 

116 

117 # Keep a reference to the new module to prevent it from dying 

118 sys.modules[patched_name] = module 

119 finally: 

120 saver.restore() # Put the original modules back 

121 

122 return module

138def patch_function(func, *additional_modules): 

139 """Decorator that returns a version of the function that patches 

140 some modules for the duration of the function call. This is 

141 deeply gross and should only be used for functions that import 

142 network libraries within their function bodies that there is no 

143 way of getting around.""" 

144 if not additional_modules: 

145 # supply some defaults 

146 additional_modules = ( 

147 _green_os_modules() + 

148 _green_select_modules() + 

149 _green_socket_modules() + 

150 _green_thread_modules() + 

151 _green_time_modules()) 

152 

153 def patched(*args, **kw): 

154 saver = SysModulesSaver() 

155 for name, mod in additional_modules: 

156 saver.save(name) 

157 sys.modules[name] = mod 

158 try: 

159 return func(*args, **kw) 

160 finally: 

161 saver.restore() 

162 return patched

165def _original_patch_function(func, *module_names): 

166 """Kind of the contrapositive of patch_function: decorates a 

167 function such that when it's called, sys.modules is populated only 

168 with the unpatched versions of the specified modules. Unlike 

169 patch_function, only the names of the modules need be supplied, 

170 and there are no defaults. This is a gross hack; tell your kids not 

171 to import inside function bodies!""" 

172 def patched(*args, **kw): 

173 saver = SysModulesSaver(module_names) 

174 for name in module_names: 

175 sys.modules[name] = original(name) 

176 try: 

177 return func(*args, **kw) 

178 finally: 

179 saver.restore() 

180 return patched

183def original(modname): 

184 """ This returns an unpatched version of a module; this is useful for 

185 Eventlet itself (i.e. tpool).""" 

186 # note that it's not necessary to temporarily install unpatched 

187 # versions of all patchable modules during the import of the 

188 # module; this is because none of them import each other, except 

189 # for threading which imports thread 

190 original_name = '__original_module_' + modname 

191 if original_name in sys.modules: 

192 return sys.modules.get(original_name) 

193 

194 # re-import the "pure" module and store it in the global _originals 

195 # dict; be sure to restore whatever module had that name already 

196 saver = SysModulesSaver((modname,)) 

197 sys.modules.pop(modname, None) 

198 # some rudimentary dependency checking -- fortunately the modules 

199 # we're working on don't have many dependencies so we can just do 

200 # some special-casing here 

201 if six.PY2: 

202 deps = {'threading': 'thread', 'Queue': 'threading'} 

203 if six.PY3: 

204 deps = {'threading': '_thread', 'queue': 'threading'} 

205 if modname in deps: 

206 dependency = deps[modname] 

207 saver.save(dependency) 

208 sys.modules[dependency] = original(dependency) 

209 try: 

210 real_mod = __import__(modname, {}, {}, modname.split('.')[:-1]) 

211 if modname in ('Queue', 'queue') and not hasattr(real_mod, '_threading'): 

212 # tricky hack: Queue's constructor in <2.7 imports 

213 # threading on every instantiation; therefore we wrap 

214 # it so that it always gets the original threading 

215 real_mod.Queue.__init__ = _original_patch_function( 

216 real_mod.Queue.__init__, 

217 'threading') 

218 # save a reference to the unpatched module so it doesn't get lost 

219 sys.modules[original_name] = real_mod 

220 finally: 

221 saver.restore() 

222 

223 return sys.modules[original_name]

228def monkey_patch(**on): 

229 """Globally patches certain system modules to be greenthread-friendly. 

230 

231 The keyword arguments afford some control over which modules are patched. 

232 If no keyword arguments are supplied, all possible modules are patched. 

233 If keywords are set to True, only the specified modules are patched. E.g., 

234 ``monkey_patch(socket=True, select=True)`` patches only the select and 

235 socket modules. Most arguments patch the single module of the same name 

236 (os, time, select). The exceptions are socket, which also patches the ssl 

237 module if present; and thread, which patches thread, threading, and Queue. 

238 

239 It's safe to call monkey_patch multiple times. 

240 """

382def _green_existing_locks(): 

383 """Make locks created before monkey-patching safe. 

384 

385 RLocks rely on a Lock and on Python 2, if an unpatched Lock blocks, it 

386 blocks the native thread. We need to replace these with green Locks. 

387 

388 This was originally noticed in the stdlib logging module.""" 

389 import gc 

390 import threading 

391 import eventlet.green.thread 

392 lock_type = type(threading.Lock()) 

393 rlock_type = type(threading.RLock()) 

394 if hasattr(threading, '_PyRLock'): 

395 # this happens on CPython3 and PyPy >= 7.0.0: "py3-style" rlocks, they 

396 # are implemented natively in C and RPython respectively 

397 py3_style = True 

398 pyrlock_type = type(threading._PyRLock()) 

399 else: 

400 # this happens on CPython2.7 and PyPy < 7.0.0: "py2-style" rlocks, 

401 # they are implemented in pure-python 

402 py3_style = False 

403 pyrlock_type = None 

404 

405 # We're monkey-patching so there can't be any greenlets yet, ergo our thread 

406 # ID is the only valid owner possible. 

407 tid = eventlet.green.thread.get_ident() 

408 for obj in gc.get_objects(): 

409 if isinstance(obj, rlock_type): 

410 if not py3_style and isinstance(obj._RLock__block, lock_type): 

411 _fix_py2_rlock(obj, tid) 

412 elif py3_style and not isinstance(obj, pyrlock_type): 

413 _fix_py3_rlock(obj)

416def _fix_py2_rlock(rlock, tid): 

417 import eventlet.green.threading 

418 old = rlock._RLock__block 

419 new = eventlet.green.threading.Lock() 

420 rlock._RLock__block = new 

421 if old.locked(): 

422 new.acquire() 

423 rlock._RLock__owner = tid

426def _fix_py3_rlock(old): 

427 import gc 

428 import threading 

429 new = threading._PyRLock() 

430 while old._is_owned(): 

431 old.release() 

432 new.acquire() 

433 if old._is_owned(): 

434 new.acquire() 

435 gc.collect() 

436 for ref in gc.get_referrers(old): 

437 try: 

438 ref_vars = vars(ref) 

439 except TypeError: 

440 pass 

441 else: 

442 for k, v in ref_vars.items(): 

443 if v == old: 

444 setattr(ref, k, new)
