9r"""Provides an API for creation of custom ClauseElements and compilers. 

10 

11Synopsis 

12======== 

13 

14Usage involves the creation of one or more 

15:class:`~sqlalchemy.sql.expression.ClauseElement` subclasses and one or 

16more callables defining its compilation:: 

17 

18 from sqlalchemy.ext.compiler import compiles 

19 from sqlalchemy.sql.expression import ColumnClause 

20 

21 class MyColumn(ColumnClause): 

22 inherit_cache = True 

23 

24 @compiles(MyColumn) 

25 def compile_mycolumn(element, compiler, **kw): 

26 return "[%s]" % element.name 

27 

28Above, ``MyColumn`` extends :class:`~sqlalchemy.sql.expression.ColumnClause`, 

29the base expression element for named column objects. The ``compiles`` 

30decorator registers itself with the ``MyColumn`` class so that it is invoked 

31when the object is compiled to a string:: 

32 

33 from sqlalchemy import select 

34 

35 s = select(MyColumn('x'), MyColumn('y')) 

36 print(str(s)) 

37 

38Produces:: 

39 

40 SELECT [x], [y] 

41 

42Dialect-specific compilation rules 

43================================== 

44 

45Compilers can also be made dialect-specific. The appropriate compiler will be 

46invoked for the dialect in use:: 

47 

48 from sqlalchemy.schema import DDLElement 

49 

50 class AlterColumn(DDLElement): 

51 inherit_cache = False 

52 

53 def __init__(self, column, cmd): 

54 self.column = column 

55 self.cmd = cmd 

56 

57 @compiles(AlterColumn) 

58 def visit_alter_column(element, compiler, **kw): 

59 return "ALTER COLUMN %s ..." % element.column.name 

60 

61 @compiles(AlterColumn, 'postgresql') 

62 def visit_alter_column(element, compiler, **kw): 

63 return "ALTER TABLE %s ALTER COLUMN %s ..." % (element.table.name, 

64 element.column.name) 

65 

66The second ``visit_alter_table`` will be invoked when any ``postgresql`` 

67dialect is used. 

68

71Compiling sub-elements of a custom expression construct 

72======================================================= 

73 

74The ``compiler`` argument is the 

75:class:`~sqlalchemy.engine.interfaces.Compiled` object in use. This object 

76can be inspected for any information about the in-progress compilation, 

77including ``compiler.dialect``, ``compiler.statement`` etc. The 

78:class:`~sqlalchemy.sql.compiler.SQLCompiler` and 

79:class:`~sqlalchemy.sql.compiler.DDLCompiler` both include a ``process()`` 

80method which can be used for compilation of embedded attributes:: 

81 

82 from sqlalchemy.sql.expression import Executable, ClauseElement 

83 

84 class InsertFromSelect(Executable, ClauseElement): 

85 inherit_cache = False 

86 

87 def __init__(self, table, select): 

88 self.table = table 

89 self.select = select 

90 

91 @compiles(InsertFromSelect) 

92 def visit_insert_from_select(element, compiler, **kw): 

93 return "INSERT INTO %s (%s)" % ( 

94 compiler.process(element.table, asfrom=True, **kw), 

95 compiler.process(element.select, **kw) 

96 ) 

97 

98 insert = InsertFromSelect(t1, select(t1).where(t1.c.x>5)) 

99 print(insert) 

100 

101Produces:: 

102 

103 "INSERT INTO mytable (SELECT mytable.x, mytable.y, mytable.z 

104 FROM mytable WHERE mytable.x > :x_1)" 

105 

106.. note:: 

107 

108 The above ``InsertFromSelect`` construct is only an example, this actual 

109 functionality is already available using the 

110 :meth:`_expression.Insert.from_select` method. 

111 

112 

113Cross Compiling between SQL and DDL compilers 

114--------------------------------------------- 

115 

116SQL and DDL constructs are each compiled using different base compilers - 

117``SQLCompiler`` and ``DDLCompiler``. A common need is to access the 

118compilation rules of SQL expressions from within a DDL expression. The 

119``DDLCompiler`` includes an accessor ``sql_compiler`` for this reason, such as 

120below where we generate a CHECK constraint that embeds a SQL expression:: 

121 

122 @compiles(MyConstraint) 

123 def compile_my_constraint(constraint, ddlcompiler, **kw): 

124 kw['literal_binds'] = True 

125 return "CONSTRAINT %s CHECK (%s)" % ( 

126 constraint.name, 

127 ddlcompiler.sql_compiler.process( 

128 constraint.expression, **kw) 

129 ) 

130 

131Above, we add an additional flag to the process step as called by 

132:meth:`.SQLCompiler.process`, which is the ``literal_binds`` flag. This 

133indicates that any SQL expression which refers to a :class:`.BindParameter` 

134object or other "literal" object such as those which refer to strings or 

135integers should be rendered **in-place**, rather than being referred to as 

136a bound parameter; when emitting DDL, bound parameters are typically not 

137supported. 

138 

139 

140Changing the default compilation of existing constructs 

141======================================================= 

142 

143The compiler extension applies just as well to the existing constructs. When 

144overriding the compilation of a built in SQL construct, the @compiles 

145decorator is invoked upon the appropriate class (be sure to use the class, 

146i.e. ``Insert`` or ``Select``, instead of the creation function such 

147as ``insert()`` or ``select()``). 

148 

149Within the new compilation function, to get at the "original" compilation 

150routine, use the appropriate visit_XXX method - this 

151because compiler.process() will call upon the overriding routine and cause 

152an endless loop. Such as, to add "prefix" to all insert statements:: 

153 

154 from sqlalchemy.sql.expression import Insert 

155 

156 @compiles(Insert) 

157 def prefix_inserts(insert, compiler, **kw): 

158 return compiler.visit_insert(insert.prefix_with("some prefix"), **kw) 

159 

160The above compiler will prefix all INSERT statements with "some prefix" when 

161compiled.

165Changing Compilation of Types 

166============================= 

167 

168``compiler`` works for types, too, such as below where we implement the 

169MS-SQL specific 'max' keyword for ``String``/``VARCHAR``:: 

170 

171 @compiles(String, 'mssql') 

172 @compiles(VARCHAR, 'mssql') 

173 def compile_varchar(element, compiler, **kw): 

174 if element.length == 'max': 

175 return "VARCHAR('max')" 

176 else: 

177 return compiler.visit_VARCHAR(element, **kw) 

178 

179 foo = Table('foo', metadata, 

180 Column('data', VARCHAR('max')) 

181 )

183Subclassing Guidelines 

184====================== 

185 

186A big part of using the compiler extension is subclassing SQLAlchemy 

187expression constructs. To make this easier, the expression and 

188schema packages feature a set of "bases" intended for common tasks. 

189A synopsis is as follows: 

190 

191* :class:`~sqlalchemy.sql.expression.ClauseElement` - This is the root 

192 expression class. Any SQL expression can be derived from this base, and is 

193 probably the best choice for longer constructs such as specialized INSERT 

194 statements. 

195 

196* :class:`~sqlalchemy.sql.expression.ColumnElement` - The root of all 

197 "column-like" elements. Anything that you'd place in the "columns" clause of 

198 a SELECT statement (as well as order by and group by) can derive from this - 

199 the object will automatically have Python "comparison" behavior. 

200 

201 :class:`~sqlalchemy.sql.expression.ColumnElement` classes want to have a 

202 ``type`` member which is expression's return type. This can be established 

203 at the instance level in the constructor, or at the class level if its 

204 generally constant:: 

205 

206 class timestamp(ColumnElement): 

207 type = TIMESTAMP() 

208 inherit_cache = True 

209 

210* :class:`~sqlalchemy.sql.functions.FunctionElement` - This is a hybrid of a 

211 ``ColumnElement`` and a "from clause" like object, and represents a SQL 

212 function or stored procedure type of call. Since most databases support 

213 statements along the line of "SELECT FROM <some function>" 

214 ``FunctionElement`` adds in the ability to be used in the FROM clause of a 

215 ``select()`` construct:: 

216 

217 from sqlalchemy.sql.expression import FunctionElement 

218 

219 class coalesce(FunctionElement): 

220 name = 'coalesce' 

221 inherit_cache = True 

222 

223 @compiles(coalesce) 

224 def compile(element, compiler, **kw): 

225 return "coalesce(%s)" % compiler.process(element.clauses, **kw) 

226 

227 @compiles(coalesce, 'oracle') 

228 def compile(element, compiler, **kw): 

229 if len(element.clauses) > 2: 

230 raise TypeError("coalesce only supports two arguments on Oracle") 

231 return "nvl(%s)" % compiler.process(element.clauses, **kw) 

232 

233* :class:`.ExecutableDDLElement` - The root of all DDL expressions, 

234 like CREATE TABLE, ALTER TABLE, etc. Compilation of 

235 :class:`.ExecutableDDLElement` subclasses is issued by a 

236 :class:`.DDLCompiler` instead of a :class:`.SQLCompiler`. 

237 :class:`.ExecutableDDLElement` can also be used as an event hook in 

238 conjunction with event hooks like :meth:`.DDLEvents.before_create` and 

239 :meth:`.DDLEvents.after_create`, allowing the construct to be invoked 

240 automatically during CREATE TABLE and DROP TABLE sequences. 

241 

242 .. seealso:: 

243 

244 :ref:`metadata_ddl_toplevel` - contains examples of associating 

245 :class:`.DDL` objects (which are themselves :class:`.ExecutableDDLElement` 

246 instances) with :class:`.DDLEvents` event hooks. 

247 

248* :class:`~sqlalchemy.sql.expression.Executable` - This is a mixin which 

249 should be used with any expression class that represents a "standalone" 

250 SQL statement that can be passed directly to an ``execute()`` method. It 

251 is already implicit within ``DDLElement`` and ``FunctionElement``. 

252 

253Most of the above constructs also respond to SQL statement caching. A 

254subclassed construct will want to define the caching behavior for the object, 

255which usually means setting the flag ``inherit_cache`` to the value of 

256``False`` or ``True``. See the next section :ref:`compilerext_caching` 

257for background. 

258

260.. _compilerext_caching:

262Enabling Caching Support for Custom Constructs 

263============================================== 

264 

265SQLAlchemy as of version 1.4 includes a 

266:ref:`SQL compilation caching facility <sql_caching>` which will allow 

267equivalent SQL constructs to cache their stringified form, along with other 

268structural information used to fetch results from the statement. 

269 

270For reasons discussed at :ref:`caching_caveats`, the implementation of this 

271caching system takes a conservative approach towards including custom SQL 

272constructs and/or subclasses within the caching system. This includes that 

273any user-defined SQL constructs, including all the examples for this 

274extension, will not participate in caching by default unless they positively 

275assert that they are able to do so. The :attr:`.HasCacheKey.inherit_cache` 

276attribute when set to ``True`` at the class level of a specific subclass 

277will indicate that instances of this class may be safely cached, using the 

278cache key generation scheme of the immediate superclass. This applies 

279for example to the "synopsis" example indicated previously:: 

280 

281 class MyColumn(ColumnClause): 

282 inherit_cache = True 

283 

284 @compiles(MyColumn) 

285 def compile_mycolumn(element, compiler, **kw): 

286 return "[%s]" % element.name 

287 

288Above, the ``MyColumn`` class does not include any new state that 

289affects its SQL compilation; the cache key of ``MyColumn`` instances will 

290make use of that of the ``ColumnClause`` superclass, meaning it will take 

291into account the class of the object (``MyColumn``), the string name and 

292datatype of the object:: 

293 

294 >>> MyColumn("some_name", String())._generate_cache_key() 

295 CacheKey( 

296 key=('0', <class '__main__.MyColumn'>, 

297 'name', 'some_name', 

298 'type', (<class 'sqlalchemy.sql.sqltypes.String'>, 

299 ('length', None), ('collation', None)) 

300 ), bindparams=[]) 

301 

302For objects that are likely to be **used liberally as components within many 

303larger statements**, such as :class:`_schema.Column` subclasses and custom SQL 

304datatypes, it's important that **caching be enabled as much as possible**, as 

305this may otherwise negatively affect performance. 

306 

307An example of an object that **does** contain state which affects its SQL 

308compilation is the one illustrated at :ref:`compilerext_compiling_subelements`; 

309this is an "INSERT FROM SELECT" construct that combines together a 

310:class:`_schema.Table` as well as a :class:`_sql.Select` construct, each of 

311which independently affect the SQL string generation of the construct. For 

312this class, the example illustrates that it simply does not participate in 

313caching:: 

314 

315 class InsertFromSelect(Executable, ClauseElement): 

316 inherit_cache = False 

317 

318 def __init__(self, table, select): 

319 self.table = table 

320 self.select = select 

321 

322 @compiles(InsertFromSelect) 

323 def visit_insert_from_select(element, compiler, **kw): 

324 return "INSERT INTO %s (%s)" % ( 

325 compiler.process(element.table, asfrom=True, **kw), 

326 compiler.process(element.select, **kw) 

327 ) 

328 

329While it is also possible that the above ``InsertFromSelect`` could be made to 

330produce a cache key that is composed of that of the :class:`_schema.Table` and 

331:class:`_sql.Select` components together, the API for this is not at the moment 

332fully public. However, for an "INSERT FROM SELECT" construct, which is only 

333used by itself for specific operations, caching is not as critical as in the 

334previous example. 

335 

336For objects that are **used in relative isolation and are generally 

337standalone**, such as custom :term:`DML` constructs like an "INSERT FROM 

338SELECT", **caching is generally less critical** as the lack of caching for such 

339a construct will have only localized implications for that specific operation. 

340

342Further Examples 

343================ 

344 

345"UTC timestamp" function 

346------------------------- 

347 

348A function that works like "CURRENT_TIMESTAMP" except applies the 

349appropriate conversions so that the time is in UTC time. Timestamps are best 

350stored in relational databases as UTC, without time zones. UTC so that your 

351database doesn't think time has gone backwards in the hour when daylight 

352savings ends, without timezones because timezones are like character 

353encodings - they're best applied only at the endpoints of an application 

354(i.e. convert to UTC upon user input, re-apply desired timezone upon display). 

355 

356For PostgreSQL and Microsoft SQL Server:: 

357 

358 from sqlalchemy.sql import expression 

359 from sqlalchemy.ext.compiler import compiles 

360 from sqlalchemy.types import DateTime 

361 

362 class utcnow(expression.FunctionElement): 

363 type = DateTime() 

364 inherit_cache = True 

365 

366 @compiles(utcnow, 'postgresql') 

367 def pg_utcnow(element, compiler, **kw): 

368 return "TIMEZONE('utc', CURRENT_TIMESTAMP)" 

369 

370 @compiles(utcnow, 'mssql') 

371 def ms_utcnow(element, compiler, **kw): 

372 return "GETUTCDATE()" 

373 

374Example usage:: 

375 

376 from sqlalchemy import ( 

377 Table, Column, Integer, String, DateTime, MetaData 

378 ) 

379 metadata = MetaData() 

380 event = Table("event", metadata, 

381 Column("id", Integer, primary_key=True), 

382 Column("description", String(50), nullable=False), 

383 Column("timestamp", DateTime, server_default=utcnow()) 

384 ) 

385 

386"GREATEST" function 

387------------------- 

388 

389The "GREATEST" function is given any number of arguments and returns the one 

390that is of the highest value - its equivalent to Python's ``max`` 

391function. A SQL standard version versus a CASE based version which only 

392accommodates two arguments:: 

393 

394 from sqlalchemy.sql import expression, case 

395 from sqlalchemy.ext.compiler import compiles 

396 from sqlalchemy.types import Numeric 

397 

398 class greatest(expression.FunctionElement): 

399 type = Numeric() 

400 name = 'greatest' 

401 inherit_cache = True 

402 

403 @compiles(greatest) 

404 def default_greatest(element, compiler, **kw): 

405 return compiler.visit_function(element) 

406 

407 @compiles(greatest, 'sqlite') 

408 @compiles(greatest, 'mssql') 

409 @compiles(greatest, 'oracle') 

410 def case_greatest(element, compiler, **kw): 

411 arg1, arg2 = list(element.clauses) 

412 return compiler.process(case((arg1 > arg2, arg1), else_=arg2), **kw) 

413 

414Example usage:: 

415 

416 Session.query(Account).\ 

417 filter( 

418 greatest( 

419 Account.checking_balance, 

420 Account.savings_balance) > 10000 

421 )

423"false" expression 

424------------------ 

425 

426Render a "false" constant expression, rendering as "0" on platforms that 

427don't have a "false" constant:: 

428 

429 from sqlalchemy.sql import expression 

430 from sqlalchemy.ext.compiler import compiles 

431 

432 class sql_false(expression.ColumnElement): 

433 inherit_cache = True 

434 

435 @compiles(sql_false) 

436 def default_false(element, compiler, **kw): 

437 return "false" 

438 

439 @compiles(sql_false, 'mssql') 

440 @compiles(sql_false, 'mysql') 

441 @compiles(sql_false, 'oracle') 

442 def int_false(element, compiler, **kw): 

443 return "0" 

444 

445Example usage:: 

446 

447 from sqlalchemy import select, union_all 

448 

449 exp = union_all( 

450 select(users.c.name, sql_false().label("enrolled")), 

451 select(customers.c.name, customers.c.enrolled) 

452 )

459def compiles(class_, *specs): 

460 """Register a function as a compiler for a 

461 given :class:`_expression.ClauseElement` type.""" 

462 

463 def decorate(fn): 

464 # get an existing @compiles handler 

465 existing = class_.__dict__.get("_compiler_dispatcher", None) 

466 

467 # get the original handler. All ClauseElement classes have one 

468 # of these, but some TypeEngine classes will not. 

469 existing_dispatch = getattr(class_, "_compiler_dispatch", None) 

470 

471 if not existing: 

472 existing = _dispatcher() 

473 

474 if existing_dispatch: 

475 

476 def _wrap_existing_dispatch(element, compiler, **kw): 

477 try: 

478 return existing_dispatch(element, compiler, **kw) 

479 except exc.UnsupportedCompilationError as uce: 

480 raise exc.UnsupportedCompilationError( 

481 compiler, 

482 type(element), 

483 message="%s construct has no default " 

484 "compilation handler." % type(element), 

485 ) from uce 

486 

487 existing.specs["default"] = _wrap_existing_dispatch 

488 

489 # TODO: why is the lambda needed ? 

490 setattr( 

491 class_, 

492 "_compiler_dispatch", 

493 lambda *arg, **kw: existing(*arg, **kw), 

494 ) 

495 setattr(class_, "_compiler_dispatcher", existing) 

496 

497 if specs: 

498 for s in specs: 

499 existing.specs[s] = fn 

500 

501 else: 

502 existing.specs["default"] = fn 

503 return fn 

504 

505 return decorate 

506 

507 

508def deregister(class_): 

509 """Remove all custom compilers associated with a given 

510 :class:`_expression.ClauseElement` type. 

511 

512 """ 

513 

514 if hasattr(class_, "_compiler_dispatcher"): 

515 class_._compiler_dispatch = class_._original_compiler_dispatch 

516 del class_._compiler_dispatcher 

517 

518 

519class _dispatcher: 

520 def __init__(self): 

521 self.specs = {} 

522 

523 def __call__(self, element, compiler, **kw): 

524 # TODO: yes, this could also switch off of DBAPI in use. 

525 fn = self.specs.get(compiler.dialect.name, None) 

526 if not fn: 

527 try: 

528 fn = self.specs["default"] 

529 except KeyError as ke: 

530 raise exc.UnsupportedCompilationError( 

531 compiler, 

532 type(element), 

533 message="%s construct has no default " 

534 "compilation handler." % type(element), 

535 ) from ke 

536 

537 # if compilation includes add_to_result_map, collect add_to_result_map 

538 # arguments from the user-defined callable, which are probably none 

539 # because this is not public API. if it wasn't called, then call it 

540 # ourselves. 

541 arm = kw.get("add_to_result_map", None) 

542 if arm: 

543 arm_collection = [] 

544 kw["add_to_result_map"] = lambda *args: arm_collection.append(args) 

545 

546 expr = fn(element, compiler, **kw) 

547 

548 if arm: 

549 if not arm_collection: 

550 arm_collection.append( 

551 (None, None, (element,), sqltypes.NULLTYPE) 

552 ) 

553 for tup in arm_collection: 

554 arm(*tup) 

555 return expr
